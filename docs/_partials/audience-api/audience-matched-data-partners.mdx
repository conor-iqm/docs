See [TypeScript Prerequisites](/Getting%20Started/Typescript-Prerequisites.mdx) for usage. 

```ts
import {
  getInstance
} from "prerequisites"

const axios = getInstance();

interface Responses {
  200: {
    content: {
      "application/json;charset=UTF-8": {
        success: boolean;
        data: {
          id: number;
          label: string;
          keyName: string;
        }[];
      };
    };
  };
}

function getMatchedAudienceDataFormats(): Promise<Responses> {
  const options = {
    method: 'GET',
    url: 'https://api.iqm.com/api/v3/audience/static/matched/data-format',
  };

  return axios.request(options).then(({ data }: { data: Responses }) => data);
}
```
