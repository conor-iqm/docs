```ts
import {
  getInstance
} from "prerequisites"

const axios = getInstance();

interface Responses {
  200: {
    content: {
      "application/json": {
        success: boolean;
        data: {
          totalRecords: number;
          prebidSegmentData: {
            providerId: number;
            providerName: string;
            description: string;
            providerLogoUrl: string;
          }[];
          filteredRecords: number;
        }
      };
    };
  };
}

function getProviders(): Promise<Responses> {
  const options = {
    method: 'GET',
    url: 'https://api.iqm.com/api/v3/audience/prebid/static/providers/list',
  }

  return axios.request(options).then(({ data }: { data: Responses }) => data);
}
```
